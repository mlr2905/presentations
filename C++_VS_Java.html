C++ VS Java

Feature C++ Java
Memory Management Direct, allows direct memory manipulation Automatic, managed by a virtual machine (JVM)
Platform Requires recompilation for each platform Write once, run anywhere with JVM support
Code Standardization No clear standardization, code needs adaptation for each platform Highly standardized, runs on any JVM-supported machine
Language Type System Pointers, complex data structures, local programming model Stable type system, based on virtual coding and execution
C++: Hello, World!

#include <iostream>
using namespace std;

int main() {
    cout << "Hello, World!" << endl;
    return 0;
}
Java: Hello, World!

public class HelloWorld {
    public static void main(String[] args) {
        System.out.println("Hello, World!");
    }
}
C++: Variables

#include <iostream>
using namespace std;

int main() {
    int num = 5;
    double decimal = 5.5;
    char letter = 'A';

    cout << "Integer: " << num << endl;
    cout << "Double: " << decimal << endl;
    cout << "Character: " << letter << endl;

    return 0;
}
Java: Variables

public class Variables {
    public static void main(String[] args) {
        int num = 5;
        double decimal = 5.5;
        char letter = 'A';

        System.out.println("Integer: " + num);
        System.out.println("Double: " + decimal);
        System.out.println("Character: " + letter);
    }
}

C++: Loops

#include <iostream>
using namespace std;

int main() {
    for(int i = 0; i < 5; i++) {
        cout << "i = " << i << endl;
    }

    int j = 0;
    while(j < 5) {
        cout << "j = " << j << endl;
        j++;
    }

    return 0;
}
Java: Loops

public class Loops {
    public static void main(String[] args) {
        for(int i = 0; i < 5; i++) {
            System.out.println("i = " + i);
        }

        int j = 0;
        while(j < 5) {
            System.out.println("j = " + j);
            j++;
        }
    }
}
C++: Arrays

#include <iostream>
using namespace std;

int main() {
    int arr[5] = {1, 2, 3, 4, 5};

    cout << "Array elements: ";
    for(int i = 0; i < 5; i++) {
        cout << arr[i] << " ";
    }
    cout << endl;

    return 0;
}
Java: Arrays

public class ArraysExample {
    public static void main(String[] args) {
        int[] arr = {1, 2, 3, 4, 5};

        System.out.print("Array elements: ");
        for(int i = 0; i < arr.length; i++) {
            System.out.print(arr[i] + " ");
        }
        System.out.println();
    }
}
C++: Objects

#include <iostream>
using namespace std;

class Car {
    public:
        string brand;
        int year;
    
        Car(string b, int y) {
            brand = b;
            year = y;
        }

        void displayInfo() {
            cout << "Brand: " << brand << ", Year: " << year << endl;
        }
};

int main() {
    Car myCar("Toyota", 2022);
    myCar.displayInfo();

    return 0;
}
Java: Objects

public class Car {
    private String brand;
    private int year;

    public Car(String b, int y) {
        brand = b;
        year = y;
    }

    public void displayInfo() {
        System.out.println("Brand: " + brand + ", Year: " + year);
    }

    public static void main(String[] args) {
        Car myCar = new Car("Toyota", 2022);
        myCar.displayInfo();
    }
}
